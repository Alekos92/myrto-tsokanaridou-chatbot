# List of regex-based replaces for answers
# Format: regex ##->## replacement

#(you) ##->## noone
(\[\][ ]*\(\)) ##->##
(\[([^\]]+)\][ ]*\((/r/[^)]+)\)) ##->## \1: https://www.reddit.com\2
(\[[^\]]*\][ ]*\((/r/[^)]+)\)) ##->## https://www.reddit.com\1
(\[([^\]]+)\][ ]*\((http[^)]+)\)) ##->## \1: \2
(\[\][ ]*\((http[^)]+)\)) ##->## \1
(\[[^\]]*\][ ]*\([^)]*\)) ##->##
